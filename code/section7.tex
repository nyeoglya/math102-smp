\subsection{$\R^2$에서의 그린 정리와 발산 정리}

\begin{theorem}[발산 정리]
만약 $F: \R^2 \to \R^2$가 정규 집합 $D$에서 $C^1$이면,
$$
\int_D \text{div} \textbf{F} dA = \int_{\partial D} \textbf{F} \cdot \textbf{N} ds 
$$
\end{theorem}
